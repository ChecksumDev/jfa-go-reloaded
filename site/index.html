Debian/Ubuntu (apt)

curl https://apt.hrfee.dev/hrfee.pubkey.gpg | sudo apt-key add -

echo "deb https://apt.hrfee.dev trusty main" | sudo tee /etc/apt/sources.list.d/hrfee.list

sudo apt-get update

sudo apt-get update

# For servers
